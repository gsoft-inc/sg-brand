<details>
    <summary >We recommend exporting your icons from <code>index.js</code> files</summary>
    ```jsx
    // assets/index.js

    import { createIcon, createMultiVariantIcon } from "@sharegate/orbit-ui";
    import { ReactComponent as Azure32 } from "./icon-azure-32.svg";
    import { ReactComponent as Filter24 } from "./icon-filter-24.svg";
    import { ReactComponent as Filter32 } from "./icon-filter-32.svg";

    export const AzureIcon32 = createIcon(Azure32);
    export const FilterMajorIcon = createMultiVariantIcon(Filter24, Filter32);
    ```

    ```jsx
    import { AzureIcon32, FilterMajorIcon } from "./assets";

    () => (
        <>
            <AzureIcon32 />
            <FilterMajorIcon />
        </>
    );
    ```
